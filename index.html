pys

Project pys is a simple(< 1 Kb) javascript for browserjs / nodejs library to engance String.substring / Array.slice with python slice style.

Install 

npm install pys

Then import it. 

// import library use script tag.
<script type="text/javascript" src="dist/pys.min.js"></script>

// or ES6 style.
var pys = require('pys');

1. Usage for javascript String

var str = '1234567890';

pys(str)('2:5');        // '345'
pys(str)(':-2');        // '12345678'
pys(str)('-2:');        // '90'
pys(str)('1:5:2');      // '24'
pys(str)('5:1:-2');     // '64'

2. Usage for javascript Array

var arr = [1, '2', 3, '4', 5, '6', 7, '8', 9, '0'];

pys(arr)('2:5');        // [3, '4', 5]
pys(arr)(':-2');        // [1, '2', 3, '4', 5, '6', 7, '8']
pys(arr)('-2:');        // [9, '0']
pys(arr)('1:5:2');      // ['2', '4']
pys(arr)('5:1:-2');     // ['6', '4']
